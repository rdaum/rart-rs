8    simd_runtime_generate!(
9        pub fn simdeez_find_insert_pos(key: u8, keys: &[u8], ff_mask_out: u32) -> Option<usize> {
10            let key_cmp_vec = S::Vi8::set1(key as i8);
11            let key_vec =
12                unsafe { SimdBaseIo::load_from_ptr_unaligned(keys.as_ptr() as *const i8) };
13            let results = key_cmp_vec.cmp_lt(key_vec);
14            let bitfield = results.get_mask() & (ff_mask_out as u32);
15            if bitfield != 0 {
16                let idx = bitfield.trailing_zeros() as usize;
17                return Some(idx);
18            }
19            None
20        }
21    );

23    simd_runtime_generate!(
24        pub fn simdeez_find_key(key: u8, keys: &[u8], ff_mask_out: u32) -> Option<usize> {
25            let key_cmp_vec = S::Vi8::set1(key as i8);
26            let key_vec =
27                unsafe { SimdBaseIo::load_from_ptr_unaligned(keys.as_ptr() as *const i8) };
28            let results = key_cmp_vec.cmp_eq(key_vec);
29            let bitfield = results.get_mask() & (ff_mask_out as u32);
30            if bitfield != 0 {
31                let idx = bitfield.trailing_zeros() as usize;
32                return Some(idx);

39#[cfg(not(feature = "simd_keys"))]
40fn binary_find_key(key: u8, keys: &[u8], num_children: usize) -> Option<usize> {
41    let mut left = 0;
42    let mut right = num_children;
43    while left < right {
44        let mid = (left + right) / 2;
45        match keys[mid].cmp(&key) {
46            std::cmp::Ordering::Less => left = mid + 1,
47            std::cmp::Ordering::Equal => return Some(mid),
48            std::cmp::Ordering::Greater => right = mid,
49        }
50    }
51    None
52}

54pub fn u8_keys_find_key_position_sorted<const WIDTH: usize>(
55    key: u8,
56    keys: &[u8],
57    num_children: usize,
58) -> Option<usize> {
59    // Width 4 and under, just use linear search.
60    if WIDTH <= 4 {
61        return (0..num_children).find(|&i| keys[i] == key);
62    }
63
64    #[cfg(feature = "simd_keys")]
65    {
66        simd_keys::simdeez_find_key(key, keys, (1 << num_children) - 1)
67    }
68
69    // Fallback to binary search.
70    #[cfg(not(feature = "simd_keys"))]
71    binary_find_key(key, keys, num_children)
72}

74pub fn u8_keys_find_insert_position_sorted<const WIDTH: usize>(
75    key: u8,
76    keys: &[u8],
77    num_children: usize,
78) -> Option<usize> {
79    #[cfg(feature = "simd_keys")]
80    {
81        simd_keys::simdeez_find_insert_pos(key, keys, (1 << num_children) - 1)
82            .or(Some(num_children))
83    }
84
85    // Fallback: use linear search to find the insertion point.
86    #[cfg(not(feature = "simd_keys"))]
87    (0..num_children)
88        .rev()
89        .find(|&i| key < keys[i])
90        .or(Some(num_children))
91}

93pub fn u8_keys_find_key_position<const WIDTH: usize, Bitset: BitsetTrait>(
94    key: u8,
95    keys: &[u8],
96    children_bitmask: &Bitset,
97) -> Option<usize> {
98    // SIMD optimized
99    #[cfg(feature = "simd_keys")]
100    {
101        // Special 0xff key is special
102        let mut mask = (1 << WIDTH) - 1;
103        if key == 255 {
104            mask &= children_bitmask.as_bitmask() as u32;
105        }
106        simd_keys::simdeez_find_key(key, keys, mask)
107    }
108
109    #[cfg(not(feature = "simd_keys"))]
110    {
111        // Fallback to linear search for non-SIMD.
112        for (i, k) in keys.iter().enumerate() {
113            if key == 255 && !children_bitmask.check(i) {
114                continue;
115            }
116            if *k == key {
117                return Some(i);
118            }
119        }
120        None
121    }
